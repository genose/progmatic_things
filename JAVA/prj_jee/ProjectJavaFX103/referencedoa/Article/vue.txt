CREATE VIEW [dbo].[VueArticle]
AS
SELECT        dbo.ARTICLE.ID_ARTICLE, dbo.ARTICLE.NOM_ARTICLE, dbo.ARTICLE.PRIX_ACHAT, dbo.ARTICLE.VOLUME, dbo.ARTICLE.TITRAGE, dbo.COULEUR.NOM_COULEUR, dbo.MARQUE.NOM_MARQUE, dbo.PAYS.NOM_PAYS, 
                         dbo.CONTINENT.NOM_CONTINENT, dbo.TYPEBIERE.NOM_TYPE, dbo.MARQUE.ID_MARQUE, dbo.PAYS.ID_PAYS, dbo.CONTINENT.ID_CONTINENT, dbo.TYPEBIERE.ID_TYPE, dbo.COULEUR.ID_COULEUR, 
                         dbo.FABRICANT.ID_FABRICANT, dbo.FABRICANT.NOM_FABRICANT
FROM            dbo.ARTICLE INNER JOIN
                         dbo.MARQUE ON dbo.ARTICLE.ID_MARQUE = dbo.MARQUE.ID_MARQUE INNER JOIN
                         dbo.PAYS ON dbo.MARQUE.ID_PAYS = dbo.PAYS.ID_PAYS INNER JOIN
                         dbo.CONTINENT ON dbo.PAYS.ID_CONTINENT = dbo.CONTINENT.ID_CONTINENT LEFT OUTER JOIN
                         dbo.FABRICANT ON dbo.MARQUE.ID_FABRICANT = dbo.FABRICANT.ID_FABRICANT LEFT OUTER JOIN
                         dbo.TYPEBIERE ON dbo.ARTICLE.ID_TYPE = dbo.TYPEBIERE.ID_TYPE LEFT OUTER JOIN
                         dbo.COULEUR ON dbo.ARTICLE.ID_COULEUR = dbo.COULEUR.ID_COULEUR
GO
